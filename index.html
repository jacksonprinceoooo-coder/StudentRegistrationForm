cmbDept.setBounds(200, 190, 250, 25); add(cmbDept); lblYear = new JLabel("Year:"); lblYear.setBounds(50, 230, 120, 25); add(lblYear); String years[] = {"I Year", "II Year", "III Year", "IV Year"}; cmbYear = new JComboBox<>(years);
